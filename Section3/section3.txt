tipo input -> UTILIZADO PARA ARGUMENTOS

mutation {
  novoUsuario(
    dados:{
    	nome: "Ronaldo",
    	email: "ronaldo@email.com",
    	idade: 24
    }
  ) {
    id nome email perfil { nome }
  }
  excluirUsuario( filtro: {
    email: "gabriel@email.com"
  }) {
    nome
  }
  alterarUsuario(
    filtro:{
    	email: "gabriel@email.com"
    }
    dados:{
    	email: "gabriel@gmail.com"
    }
  ) {
    id nome email idade perfil { nome }
  }
}
